import networkx as nx

def has_bridges(edge_list, root):
    """
    Decide whether the connected component containing `root` has any bridges.

    Parameters
    ----------
    edge_list : list
        Unweighted undirected edges as [u, v] (or weighted as [u, v, w], which will be ignored).
    root : hashable
        Root node. If specified, only bridges in the connected component containing this node
        are considered.

    Returns
    -------
    bool
        Whether the connected component containing `root` has any bridges.
    """
    G = nx.Graph()

    for e in edge_list:
        if len(e) >= 2:
            u, v = e[0], e[1]
            G.add_edge(u, v)

    # Strictly following docs: has_bridges(G, root=None)
    result = nx.has_bridges(G, root=root)
    return result
